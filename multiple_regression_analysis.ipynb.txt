    "# Variance Inflation Factor (VIF) — Check Multicollinearity\n",
    "feature_cols = ['TV_encoded', 'Radio', 'Social Media',\n",
    "                'Influencer_Mega', 'Influencer_Micro', 'Influencer_Nano']\n",
    "\n",
    "# Keep only bool/numeric\n",
    "bool_cols = [c for c in feature_cols if c in df_clean.columns]\n",
    "X_vif = df_clean[bool_cols].astype(float)\n",
    "\n",
    "vif_data = pd.DataFrame()\n",
    "vif_data['Feature'] = X_vif.columns\n",
    "vif_data['VIF'] = [variance_inflation_factor(X_vif.values, i) \n",
    "                   for i in range(X_vif.shape[1])]\n",
    "\n",
    "print(\"VIF Results:\")\n",
    "print(vif_data)\n",
    "print(\"\\nInterpretation:\")\n",
    "print(\"VIF < 5  => No multicollinearity (Good)\")\n",
    "print(\"VIF 5-10 => Moderate multicollinearity\")\n",
    "print(\"VIF > 10 => Severe multicollinearity (Problem)\")\n"
   ]

    "# Prepare features and target\n",
    "feature_cols = ['TV_encoded', 'Radio', 'Social Media',\n",
    "                'Influencer_Mega', 'Influencer_Micro', 'Influencer_Nano']\n",
    "available = [c for c in feature_cols if c in df_clean.columns]\n",
    "\n",
    "X = df_clean[available].astype(float)\n",
    "y = df_clean['Sales']\n",
    "\n",
    "# Add constant\n",
    "X_const = sm.add_constant(X)\n",
    "\n",
    "# Fit OLS model\n",
    "model = sm.OLS(y, X_const).fit()\n",
    "print(model.summary())\n"
   ]
